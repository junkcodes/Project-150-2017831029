#include <algorithm>  
#include <iostream> 
using namespace std; 
  
int main() 
{ 
    int t,c=1;
    scanf("%d",&t);
    while(t--){
        int n,k;
        scanf("%d %d",&n,&k);
        int arr[n];
        for(int i=0; i < n; i++)
            arr[i]=i;
        printf("Case %d:\n",c);
        c++;
        do {
            for(int i=0; i < n; i++)
                printf("%c",arr[i]+'A');
            printf("\n");
        } while (next_permutation(arr, arr + n) && --k); 
    }
}
